function cdwe
  set old_dir $PWD

  {{{cd_command}}} $argv; or return

  set new_dir $PWD

  set result ({{{exec_path}}} run --old_dir="$old_dir" --new_dir="$new_dir")
  eval $result
end

function cdwe-remove
  ({{{exec_path}}} remove fish)
end

function cdwe-reload
  ({{{exec_path}}} reload fish)
end
