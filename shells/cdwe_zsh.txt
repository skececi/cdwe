function cdwe() {
  local old_dir="$PWD"

  {{{cd_command}}} "$@" || return

  local new_dir="$PWD"

  local result
  result="$({{{exec_path}}} run --old_dir="$old_dir" --new_dir="$new_dir")"
  eval "${result}"
}

function cdwe-reload () {
  {{{exec_path}}} reload zsh
}

function cdwe-remove() {
  {{{exec_path}}} remove zsh 
}
